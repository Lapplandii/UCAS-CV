% 定义其他格式
\newcommand{\datedsection}[2]{      % #1是标题，#2是日期
  \section[#1]{#1 \hfill #2}        % [#1] 用于生成目录，{#1 \hfill #2} 在页面显示标题，% \hfill 自动填满中间空格
}
\newcommand{\datedsubsection}[2]{   % 二级标题
  \subsection[#1]{#1 \hfill #2}     % #1 通常是“单位/学校”，#2 是“起止时间”。
}
\newcommand{\datedline}[2]{         % 定义一个名为 \datedline 的新命令，接受 2 个参数
  {\par #1 \hfill #2 \par}          % \par 表示开启一个新段落。在这一行内，左边放内容 #1，右边放日期 #2。
                                    % 两端的 \par 确保这一行独立存在，不会与前后文字混在一起。
}
\newcommand{\name}[1]{ 
  \centerline{                      % \centerline 使内容居中；\Huge 设置最大号字；\bfseries 设置加粗
    \Huge\bfseries
    \xeCJKsetup{CJKglue=\hspace{0.2em plus 0.1em}} % 临时设置汉字之间的间距为0.3
    #1
  }
  \xeCJKsetup{CJKglue=\hspace{0pt}} % 恢复默认间距，否则全文都会变松散
}
\newcommand{\contactInfo}[2]{       % #1邮箱，#2手机
  \centerline{\large{ {\color{myblue}\faEnvelope}~{#1} \textperiodcentered\ \ {\color{myblue}\faPhone}~{#2}}}
}
\newcommand{\role}[2]{                 % #1职位/动作，#2具体描述
  {\par \textit{#1} ~ #2 \par}         % \par 确保独立成段；\textit{#1} 将职位设为斜体
  \vspace{0.5ex}                       
}

% 根据是否填写了内容自动决定是否显示分隔圆点
\newcommand{\otherInfo}[4]{            % 定义接受 4 个参数的宏（例如：电话、邮箱、GitHub、领英）
  \centerline{\large{ {#1}}            % 居中显示第 1 个必填参数
    \ifthenelse{\isempty{#2}}%         % 如果第 2 个参数为空
      { }                              % 则什么都不做
      {\textperiodcentered\ \ {#2} }   % 如果不为空，则先画一个圆点再显示参数内容
    \ifthenelse{\isempty{#3}}%         % 同样判断第 3 个参数
      { }                              % 为空则跳过
      {\textperiodcentered\ \ {#3} }   % 不为空则显示圆点和内容
    \ifthenelse{\isempty{#4}}%         % 判断第 4 个参数
      { } 
      {\textperiodcentered\ \ {#4} }
  }
  \vspace{0.7ex}                       
}
